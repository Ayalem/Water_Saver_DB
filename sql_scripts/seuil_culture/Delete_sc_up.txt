CREATE OR REPLACE FUNCTION DELETE_SEUIL_CULTURE(
    p_seuil_id IN NUMBER
) RETURN BOOLEAN
IS
BEGIN
    DELETE FROM SEUIL_CULTURE
    WHERE seuil_id = p_seuil_id;
    
    COMMIT;
    RETURN TRUE;
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE;
END DELETE_SEUIL_CULTURE;
/
